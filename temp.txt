Terminal Tab 1: Live Production (PM2)
bash
# One-time setup
npm install pm2 -g  # or use local: npm install

# Start live production
npm run live

# Or step by step:
npm run env:live    # Switch to live environment
npm run build:live  # Build for production
npm run live:start  # Start with PM2

# Monitoring
npm run live:logs   # View logs
npm run live:monit  # Monitor in dashboard
Terminal Tab 2: Development (Test Environment)
bash
# Start development server
npm run test:dev

# Or simply:
npm run dev:test
Essential PM2 Commands Cheat Sheet
bash
# Quick start live environment
npm run live

# Development workflow
npm run test:dev

# Management commands
npm run live:stop      # Stop live
npm run live:restart   # Restart live
npm run pm2:list       # Show all processes
npm run pm2:logs       # Show all logs

# Database operations
npm run db:push:test   # Push schema to test DB
npm run db:push:live   # Push schema to live DB
